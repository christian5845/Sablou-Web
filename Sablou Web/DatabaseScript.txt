CREATE TABLE [dbo].[Bruger] (
    [Id]            INT            NOT NULL,
    [Navn]          NVARCHAR (50)  NOT NULL,
    [Email]         NVARCHAR (50)  NOT NULL,
    [Adresse]       NVARCHAR (50)  NOT NULL,
    [Telefonnummer] INT            NULL,
    [Password]      NVARCHAR (200) NOT NULL,
    [Rolle]         NVARCHAR (50)  NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[Chokolade] (
    [Id]          INT             NOT NULL,
    [Navn]        NVARCHAR (30)   NOT NULL,
    [Stykpris]    DECIMAL (18, 2) NOT NULL,
    [Beskrivelse] NVARCHAR (200)  NOT NULL,
    [Antal]       INT             NOT NULL,
    PRIMARY KEY CLUSTERED ([ID] ASC)
);


CREATE TABLE [dbo].[Højtider] (
    [Id]   INT        NOT NULL,
    [Navn] NCHAR (20) NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[HøjtidsKatalog] (
    [Id]     INT NOT NULL,
    [Højtid] INT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC),
    CONSTRAINT [FK_Højtid] FOREIGN KEY ([Højtid]) REFERENCES [dbo].[Højtider] ([Id])
);

CREATE TABLE [dbo].[ChokoladerIKatalog] (
    [Id]          INT NOT NULL,
    [KatalogId]   INT NOT NULL,
    [ChokoladeId] INT NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC),
    CONSTRAINT [FK_KatalogId] FOREIGN KEY ([KatalogId]) REFERENCES [dbo].[HøjtidsKatalog] ([Id]),
    CONSTRAINT [FK_ChokoladeIKatalogId] FOREIGN KEY ([ChokoladeId]) REFERENCES [dbo].[Chokolade] ([Id])
);

CREATE TABLE [dbo].[Ingrediens] (
    [Id]          INT            NOT NULL,
    [Navn]        NVARCHAR (30)  NOT NULL,
    [Beskrivelse] NVARCHAR (100) NULL,
    [Antal]       INT            NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[IngrediensListe] (
    [Id]           INT NOT NULL,
    [ChokoladeId]  INT NOT NULL,
    [IngrediensId] INT NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC),
    CONSTRAINT [FK_ChokoladeId] FOREIGN KEY ([ChokoladeId]) REFERENCES [dbo].[Chokolade] ([Id]),
    CONSTRAINT [FK_IngrediensId] FOREIGN KEY ([IngrediensId]) REFERENCES [dbo].[Ingrediens] ([Id])
);
GO

CREATE TABLE [dbo].[Kurv] (
    [Id]       INT NOT NULL,
    [BrugerID] INT NOT NULL,
    PRIMARY KEY CLUSTERED ([ID] ASC),
    CONSTRAINT [FK_Kurv_Bruger] FOREIGN KEY ([BrugerID]) REFERENCES [dbo].[Bruger] ([Id]) ON DELETE CASCADE
);

CREATE TABLE [dbo].[Ordre] (
    [Id]          INT            NOT NULL,
    [BrugerId]    INT            NULL,
    [Navn]        NVARCHAR (100) DEFAULT ('') NOT NULL,
    [Email]       NVARCHAR (100) DEFAULT ('') NOT NULL,
    [Telefon]     NVARCHAR (20)  NULL,
    [Adresse]     NVARCHAR (200) DEFAULT ('') NOT NULL,
    [Besked]      NVARCHAR (500) NULL,
    [ErLoggetInd] BIT            DEFAULT ((0)) NOT NULL,
    [Dato]        DATETIME       DEFAULT (getdate()) NOT NULL,
    [Behandlet]   BIT            DEFAULT ((0)) NOT NULL,
	[ErAnnulleret] BIT            DEFAULT ((0)) NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[KurvLinje] (
    [ID]          INT NOT NULL,
    [KurvID]      INT NULL,
    [ChokoladeID] INT NOT NULL,
    [Antal]       INT NOT NULL,
    PRIMARY KEY CLUSTERED ([ID] ASC),
    CONSTRAINT [FK_KurvLinje_Kurv] FOREIGN KEY ([KurvID]) REFERENCES [dbo].[Kurv] ([ID]) ON DELETE CASCADE
);

CREATE TABLE [dbo].[OrdreLinje] (
    [Id]          INT NOT NULL,
    [OrdreID]     INT NULL,
    [ChokoladeID] INT NOT NULL,
    [Antal]       INT NOT NULL,
    PRIMARY KEY CLUSTERED ([ID] ASC),
    CONSTRAINT [FK_OrdreLinje_Ordre] FOREIGN KEY ([OrdreID]) REFERENCES [dbo].[Ordre] ([Id]) ON DELETE CASCADE,
CONSTRAINT [FK_OrdreLinje_Chokolade] FOREIGN KEY ([ChokoladeID]) REFERENCES [dbo].[Chokolade] ([ID]) ON DELETE CASCADE
);
